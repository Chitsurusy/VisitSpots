<script>
    /***************************************************
     *  ⚙️ 超级配置区 (管理员在这里修改) ⚙️
     *  修改完后，保存文件并重新上传到 Vercel 即可

    const CONFIG = {
        // 1. 首页配置
        homeTitle: "塱头古村·探秘之旅",  // 首页大标题
        homeDesc: "穿行古巷，寻访历史，集齐印章赢好礼", // 首页副标题
        // 首页背景图 (建议用宽图，可以是网络链接或本地文件名如 banner.jpg)
        homeBannerImg: "https://picsum.photos/seed/langtou/800/600", 
        
        // 2. 地图配置
        // 地图背景图 (建议上传一张古村的平面图，如果没有，可以使用下面这个默认色块)
        // 如果你有图，把地址填在这里，比如 "map.jpg"
        mapBackgroundImg: "https://picsum.photos/seed/mapbg/800/600", 

        // 3. 打卡点配置
        // x, y 是大头针在地图上的位置 (百分比)，例如 top:10%, left:20%
        // img 是印章图片
        spots: [
            { id: 1, name: "村口牌坊",   x: 10, y: 20, img: "https://picsum.photos/seed/s1/100/100" },
            { id: 2, name: "青云桥",     x: 30, y: 40, img: "https://picsum.photos/seed/s2/100/100" },
            { id: 3, name: "渔隐公祠",   x: 50, y: 60, img: "https://picsum.photos/seed/s3/100/100" },
            { id: 4, name: "古榕树",     x: 70, y: 30, img: "https://picsum.photos/seed/s4/100/100" },
            { id: 5, name: "积墨楼",     x: 85, y: 75, img: "https://picsum.photos/seed/s5/100/100" }
        ]
    };

    /***************************************************
     *  以下为程序逻辑，无需修改
     ***************************************************/
    
    // 状态管理
    let collectedIds = JSON.parse(localStorage.getItem('langtou_ids')) || [];
    let currentSpotId = null;

    // 初始化
    window.onload = function() {
        // 应用配置
        document.getElementById('home-title').innerText = CONFIG.homeTitle;
        document.getElementById('home-desc').innerText = CONFIG.homeDesc;
        document.getElementById('home-banner').style.backgroundImage = `url('${CONFIG.homeBannerImg}')`;
        document.getElementById('view-map').style.backgroundImage = `url('${CONFIG.mapBackgroundImg}')`;
        
        checkUrl(); // 检查是否扫码进入
        updateProgress();
    };

    // 跳转到游戏页
    function goToGame() {
        document.getElementById('page-home').classList.remove('active');
        document.getElementById('page-game').classList.add('active');
        renderList();
        renderMap();
    }

    // 切换列表/地图视图
    function switchView(type) {
        document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
        event.target.classList.add('active');
        
        if(type === 'list') {
            document.getElementById('view-list').style.display = 'block';
            document.getElementById('view-map').style.display = 'none';
        } else {
            document.getElementById('view-list').style.display = 'none';
            document.getElementById('view-map').style.display = 'block';
        }
    }

    // 渲染列表
    function renderList() {
        const listEl = document.getElementById('view-list');
        listEl.innerHTML = '';
        CONFIG.spots.forEach(spot => {
            const isDone = collectedIds.includes(spot.id);
            const div = document.createElement('div');
            div.className = `spot-card ${isDone ? 'done' : ''}`;
            div.innerHTML = `
                <div class="spot-info">
                    <h3 style="margin:0;">${spot.name}</h3>
                    <div class="spot-status">${isDone ? '✅ 已打卡' : '📍 未打卡'}</div>
                </div>
                <button class="btn-scan" onclick="openCheckin(${spot.id})" ${isDone ? 'disabled' : ''}>
                    ${isDone ? '完成' : '打卡'}
                </button>
            `;
            listEl.appendChild(div);
        });
    }

    // 渲染地图大头针
    function renderMap() {
        const mapEl = document.getElementById('view-map');
        mapEl.innerHTML = ''; // 清空
        CONFIG.spots.forEach(spot => {
            const isDone = collectedIds.includes(spot.id);
            const pin = document.createElement('div');
            pin.className = `map-pin ${isDone ? 'done' : ''}`;
            pin.style.top = spot.y + '%';
            pin.style.left = spot.x + '%';
            pin.innerHTML = `<span>${isDone ? '✔' : '●'}</span>`;
            pin.onclick = () => openCheckin(spot.id);
            mapEl.appendChild(pin);
        });
    }

    // 打开打卡弹窗
    function openCheckin(id) {
        if (collectedIds.includes(id)) {
            alert("您已经打过这个卡了！");
            return;
        }
        currentSpotId = id;
        const spot = CONFIG.spots.find(s => s.id === id);
        document.getElementById('modal-title').innerText = "打卡：" + spot.name;
        document.getElementById('stamp-preview').src = spot.img;
        document.getElementById('modal-checkin').style.display = 'flex';
    }

    function closeModal() {
        document.getElementById('modal-checkin').style.display = 'none';
    }

    // 确认打卡
    function confirmStamp() {
        if (!collectedIds.includes(currentSpotId)) {
            collectedIds.push(currentSpotId);
            localStorage.setItem('langtou_ids', JSON.stringify(collectedIds));
            updateProgress();
            renderList();
            renderMap();
            alert("打卡成功！印章已收集");
        }
        closeModal();
    }

    // 更新进度条
    function updateProgress() {
        const percent = Math.round((collectedIds.length / CONFIG.spots.length) * 100);
        document.getElementById('progress-fill').style.width = percent + '%';
        document.getElementById('progress-text').innerText = `${collectedIds.length}/${CONFIG.spots.length}`;
    }

    // 检查URL参数 (扫码功能)
    function checkUrl() {
        const params = new URLSearchParams(window.location.search);
        const spotId = params.get('spot');
        if (spotId) {
            // 如果有参数，自动跳转到游戏页并打开弹窗
            setTimeout(() => {
                goToGame();
                openCheckin(parseInt(spotId));
                // 清除参数防止刷新重复弹
                window.history.replaceState({}, document.title, window.location.pathname);
            }, 100);
        }
    }
